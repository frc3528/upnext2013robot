016public class Feeder extends Subsystem implements PneumaticDevice {

024       feeder = new PneumaticHelper(RobotMap.FEEDER_PUSH_SOLENOID_CHANNEL, 
025               RobotMap.FEEDER_PULL_SOLENOID_CHANNEL, 
026               RobotMap.FEEDER_PULL_SOLENOID_CHANNEL);
027       
028       holder = new PneumaticHelper(RobotMap.HOLDER_PUSH_SOLENOID_CHANNEL, 
029               RobotMap.HOLDER_PULL_SOLENOID_CHANNEL, 
030               RobotMap.HOLDER_PUSH_SOLENOID_CHANNEL);
031       
032   }
033   
034   public void initDefaultCommand() {
035        // Set the default command for a subsystem here.
036        //setDefaultCommand(new MySpecialCommand());
037    }
038   
039   public void pushFeeder() {
040       feeder.push();
041   } 
042   
043   public void pullFeeder() {      
044       feeder.pull();
045   }
046   
047   public void pushHolder() {    
048       holder.push();
049   }
050   
051   public void pullHolder() {     
052       holder.pull();
053   }
054   
055   public void zeroSolenoids() {
056       feeder.reset();
057       holder.reset();
058   }
059
