005package com.teamupnext.robot.commands;
006
007import com.teamupnext.robot.RobotMap;
008
009/**
010 *
011 * @author jousley
012 */
013public class BigHopShooter extends CommandBase {
014    
015    public BigHopShooter() {
016        //requires(shooter);
017    }
018    
019    protected void initialize() {       
020       shooter.setPower(RobotMap.SHOOTER_PREHOP_SPEED);
021       setTimeout(RobotMap.SHOOTER_FROM_STOP_TIMEOUT);
022    }
023
024    protected void execute() {
025        if(timeSinceInitialized() >= RobotMap.SHOOTER_PREHOP_TIMEOUT 
026                && shooter.getPower() != RobotMap.SHOOTER_HOP_SPEED) {
027            shooter.setPower(RobotMap.SHOOTER_HOP_SPEED);
028        }
029    }
030
031    protected boolean isFinished() {
032        return isTimedOut();
033    }
034
035    protected void end() {
036        shooter.setPower(RobotMap.SHOOTING_POWER_DEFAULT);
037    }
038
039    protected void interrupted() {
040    }
041}
